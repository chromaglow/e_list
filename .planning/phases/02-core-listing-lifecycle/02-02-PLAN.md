import { sqliteTable, text, integer } from 'drizzle-orm/sqlite-core'

export const listings = sqliteTable('listings', {
  id:           text('id').primaryKey(),
  title:        text('title').notNull(),
  description:  text('description').notNull(),
  price:        text('price'),
  poster_name:  text('poster_name').notNull(),
  contact_info: text('contact_info').notNull(),
  photo_key:    text('photo_key'),
  edit_token:   text('edit_token').notNull(),
  status:       text('status', { enum: ['active', 'taken', 'deleted'] }).notNull().default('active'),
  created_at:   integer('created_at', { mode: 'timestamp' }).notNull().$defaultFn(() => new Date()),
  updated_at:   integer('updated_at', { mode: 'timestamp' }).notNull().$defaultFn(() => new Date()).$onUpdateFn(() => new Date()),
  taken_at:     integer('taken_at',   { mode: 'timestamp' }),
  deleted_at:   integer('deleted_at', { mode: 'timestamp' }),
})

export type Listing    = typeof listings.$inferSelect
export type NewListing = typeof listings.$inferInsert

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Client → POST /[token]/api/listings | Untrusted JSON body crosses from browser into Route Handler |
| Route Handler → Turso (Drizzle) | Parameterized queries prevent SQL injection |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02-02-A | Spoofing | editToken field in request body | mitigate | Zod schema does not include `editToken` field — any submitted editToken is ignored; server always generates via `crypto.randomUUID()` |
| T-02-02-B | Tampering | photoKey field — forged blob pathname | accept | Zod validates length (max 500); invite gate is the primary trust boundary — a trusted friend group user submitting a fake pathname gets a broken image, no privilege escalation |
| T-02-02-C | Tampering | SQL injection via listing fields | mitigate | Drizzle ORM uses parameterized queries for all inserts — no string concatenation into SQL |
| T-02-02-D | Tampering | Oversized payloads causing OOM | mitigate | Zod enforces field max lengths: title 200, description 2000, price 100, posterName 100, contactInfo 200, photoKey 500 |
| T-02-02-E | Information Disclosure | editToken logged or echoed | mitigate | Route never logs editToken; response only returns `{ id, editToken }` directly to the posting client — confirmed in tests |
| T-02-02-SC | Tampering | npm/pip/cargo installs | accept | No new packages installed in this plan — all deps (zod, nanoid, drizzle-orm) are pre-existing Phase 1 installs |
</threat_model>
